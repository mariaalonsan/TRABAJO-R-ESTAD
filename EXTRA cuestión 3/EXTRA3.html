# Creamos un gr√°fico de barras que muestra las ventas globales por g√©nero
videojuegos %>%
  ggplot(aes(x = Genre, y = Global_Sales, fill = Genre)) +
  geom_bar(stat = "identity", color = "#000000") +
  theme_minimal() +
  labs(title = "Ventas globales por g√©nero", x = "G√©nero", y = "Ventas globales") +
  theme(legend.position = "none", plot.title = element_text(hjust = 0.5)) +
  scale_fill_brewer(palette = "Set3")

# Creamos un gr√°fico de puntos que muestra las ventas globales por a√±o
videojuegos %>%
  ggplot(aes(x = Year, y = Global_Sales, color = Genre)) +
  geom_point(alpha = 0.6, size = 2, shape = 21, fill = "#545050") +
  theme_minimal() +
  labs(title = "Ventas globales por a√±o", x = "A√±o", y = "Ventas globales") +
  theme(plot.title = element_text(hjust = 0.5)) +
  scale_color_brewer(palette = "Set1")
